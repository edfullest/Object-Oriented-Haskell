class Humano{
    [+] Integer edad;
    [+] Integer z;
    [+] Integer z3;
    [+] Integer z1;
    [+] Integer z2;
    [+] speak => Nothing{
        displayLn("Human");
    }
}

class Persona : Humano{
    [+] Integer hijos;
    [+] Integer hijos1;
    [+] Integer hijos2;
    [+] Integer hijos3;
    [+] speak => Nothing{
        displayLn("Person");
    }
}

    [+] Integer materias5;
    [+] Integer materias4;
    [+] Integer materias3;
