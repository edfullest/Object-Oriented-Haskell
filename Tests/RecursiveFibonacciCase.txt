fibonacci => Integer :: Integer n {
    case n of 
        0 => 
            return 0;
        end
        1 => 
            return 1;
        end
        otherwise => 
            return (fibonacci(n - 1) + fibonacci(n - 2));
        end
}
                                                        
main{
    Integer i = 3;
    displayLn(1,fibonacci(fibonacci(i)));
    displayLn(5,fibonacci(fibonacci(i + 2)));
    displayLn(21,fibonacci(fibonacci(i * 2)));
    displayLn(21,fibonacci(16 / 2));

}
