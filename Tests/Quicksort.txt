quicksort => Nothing :: Int low -> Int high -> Int[10] n{
    
    Int i,j;
    i = low;
    j = high;

    Int pivot;
    pivot = n[low + (high - low)/2];

    while(i<=j){
        while(n[i] < pivot){
            i++;
        }

        while(n[j] > pivot){
            j--;
        }

        if(i<=j){
           Int temp;
           temp = n[i];
           n[i] = n[j];
           n[j] = temp;
           i++;
           j--;
        }
    }
    if(low<j){
        quicksort(low,j,n);
    }
    if(i<high){
        quicksort(i,high,n);
    }
}
