class Clase{
    [+] Integer i;
    [+] Integer [2][5] matriz;
}

class Point{
    [+] Decimal x;
    [+] Decimal y;
    [+] Clase op;


    [-] callMethod => Nothing{
        Integer [2][5] mat = [[1,2,3,4,5],
                                [6,7,8,9,99999999999]];
        display("Llamada interna");
        displayLn("EEEE", x,y);
        x = 11.1;
        y = 12.1;
        op.i = 100;
        op.matriz = mat;
        displayLn("ZZZZ",x,y);
    }




    [+] editX => Decimal [2][2]{
        x = 99.99;
        Decimal [2][2] d = [[3948400990.1]];
        return d;
    }

    [+] setPoint => Nothing :: Decimal dX -> Decimal dY{
        x = dX;
        y = dY;
        
        callMethod();
        editX();
        displayLn(x,y);
        Decimal [2][2] res;
        res = editX();
        x = res[0][0];
        displayLn("X actualizao",x,y,res[0][0]);

    }



}

class Point3D : Point{
    [+] Decimal z;

    Point3D :: Decimal dX -> Decimal dY -> Decimal dZ{
        x = dX;
        y = dY;
        z = dZ;

    }


    [+] editZ => Decimal{
        z = 0.00000000001;
        return 3948400990.1;
    }

    [+] callMethod => Nothing{
        displayLn("Es publico");
        displayLn("Es publico 2");
        z = editZ();

    }



    [+] setPoint => Nothing :: Decimal dX -> Decimal dY -> Decimal dZ{
        x = dX;
        y = dY;
        z = dZ;

        callMethod();
        displayLn("Sup");

    }    
}
