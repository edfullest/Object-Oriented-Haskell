class Clase{
    [+] Integer i;
    [+] Integer [2][5] matriz;
}

class Point{
    [+] Decimal x;
    [+] Decimal y;
    [+] Clase op;


    [-] callMethod => Nothing{
        Integer [2][5] mat = [[1,2,3,4,5],
                                [6,7,8,9,99999999999]];
        display("Llamada interna");
        displayLn("EEEE", x,y);
        x = 11.1;
        y = 12.1;
        op.i = 100;
        op.matriz = mat;
        displayLn("ZZZZ",x,y);
    }
    
    [+] editX => Decimal [2][2]{
        x = 99.99;
        Decimal [2][2] d = [[292929.1,929292.2]];
        return d;
    }

    [+] setPoint => Nothing :: Decimal dX -> Decimal dY{
        x = dX;
        y = dY;
        
        callMethod();
        editX();
        displayLn(x,y);
        Decimal [2][2] res;
        res = editX();
        x = res[0][1];
        displayLn("X actualizao",x,y,"RES[0][0]",res[0][0]);

    }

    [+] getDistance => Decimal{
        return (x^2.0 + y^2.0)^0.5;
    }    
}

class Line{
    [+] Point p1;
    [+] Point p2;

    Line :: Point p11 -> Point p22{
        p1 = p11;
        p2 = p22;
    }

    [+] getLength => Decimal{
        Decimal x1,x2,y1,y2;
        x1 = p1.x;
        x2 = p2.x;
        y1 = p1.y;
        y2 = p2.y;

        return ((x2 - x1)^2.0 + (y2 - y1)^2.0)^0.5;
    }



 }
