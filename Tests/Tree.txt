class Node{
    [-] Node left;
    [-] Node right;
    [+] Integer val;

    [+] setLeftNode => Nothing :: Node l{
        left = l;
    }
    [+] setRightNode => Nothing :: Node r{
        right = r;
    }

    [+] getLeftNode => Node {
        return left;
    }
    [+] getRightNode => Node {
        return right;
    }


}

class Tree{
    [-] Node root;
    [-] Node null;
    [+] Integer sum;
    Tree :: Node n{
        root = n;
        sum = 0;
    }

    [+] setRoot => Nothing :: Node r{
        sum = 0;
        root = r;
    }

    [-] updateSum => Nothing :: Integer val{
        sum = sum + val;
    } 

    [-] preorder => Nothing :: Node n{
        if (n != null){
            Node l,r;
            l = n.getLeftNode();
            r = n.getRightNode();
            displayLn(n.val);
            Integer val;
            val = n.val;
            updateSum(val);
            preorder(l);
            preorder(r);
        }
    }



    [+] printPreorder => Nothing {
        preorder(root);
        displayLn("Finished preorder");
    }
}
