class Humano{
    [+] Decimal altura;
    [+] Integer edad;

    [+] setEdad => Nothing :: Int iEdad{
        edad = iEdad;
    }
}

class Persona : Humano{
    [+] Money sueldo;
    [+] String nombre;
    [+] Humano h;
    

    Persona :: Money mSueldo -> String sNombre {
        sueldo = mSueldo;
        nombre = sNombre;
        altura = 178.5;
        setEdad(23);
    }

    [+] setEdad => Nothing :: Int iEdad{
        edad = iEdad + 1;
    }

    [+] setSueldo => Nothing :: Money mSueldo{
        sueldo = mSueldo;
    }

    [+] setNombre => Nothing :: String sNombre{
        nombre = sNombre;
    }

    [+] getNombre => String{
        return nombre;
    }

    [+] getSueldo => Money{
        return sueldo;
    }

}

class Profesor : Persona{
    [+] String isElda; 
}

crearPersona => Persona :: Money sueldo -> String nombre{
    Persona p = Persona(sueldo,nombre);
    return p;
}


crearPersonas => Persona [10] :: Money sueldo -> String nombre{
    Persona [10] personas;
    String [10] strings = ["a","b","c","d","e","f","g","h","i","k"];
    for i in [0..9]{
        // Aquí construimos un objeto de tipo profesor
        // que usa el constructor Persona
        Profesor p = Persona(sueldo ^ i,nombre +" " + ">" +  strings[i] + "<");
        // Se lo asignamos al índice del arreglo de tipo Persona
        personas[i] = p;
    }
    return personas;
}

    Persona p;
    Persona [10] personas;
    p = crearPersona(12.2 * 100,"Eduardo");
    displayLn(p);

}
